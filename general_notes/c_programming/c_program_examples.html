C Programming Examples

Hello World

#include <stdio.h>
 
int main() {
  printf("Hello world\n");
  return 0;
}

OR

#include <stdio.h>

void main(void) {
    printf("Hello world!\n");
}

Enter an Integer and Print to Screen


#include <stdio.h>
 
int main()
{
  int a;
 
  printf("Enter an integer\n");
  scanf("%d", &a);
 
  printf("Integer that you have entered is %d\n", a);
 
  return 0;
}

If Statements


#include <stdio.h>
 
void main(int argc, char *argv[]) {
    if (argc < 2) {
        printf("You need to enter at least 1 parameter\n");
    } else if (argc < 4) {
        printf("Everything is OK.\n");
    } else {
        printf("You entered too many parameters.\n");
    }
}

Odd or Even

#include <stdio.h>

int main()
{
   int n;
 
   printf("Enter an integer\n");
   scanf("%d", &n);
 
   if (n%2 == 0)
      printf("Even\n");
   else
      printf("Odd\n");
 
   return 0;
}

For Loops

#include <stdio.h>

void main(int argc, char *argv[]) {
    int i;
    for (i=0; i < argc; i=i+1) {
        printf("Argument %d is %s\n", i, argv[i]);
    }
}

While Loops

#include <stdio.h>

void main(int argc, char *argv[]) {
    int i;
    i=0; // INITIALIZE
    while (i < argc) {
        printf("Argument %d is %s\n", i, argv[i]);
        i = i+1; // ITERATE
    }
}

Useful Loop Statements

  • Break - used to prematurely terminate a loop
  • Continue - proceeds to the next iteration of a loop, skipping the remaining statements
  • Return - used to immediately exit from a function and return to the function's caller
