Javascript Objects

Creating Objects

 literal...
        
            var car = { mileage: 0 }  //prototype is Object {}
        
    
 using Object.create()...
         
            var myCar = Object.create(car)  //prototype is whatever was inherited
        
    
 using constructor...
        
            var herAccount = new Object()  //prototype is Object{}

Giving Initial Properties to Objects

 one way...
        
            function car(make) {
                this.make = make;
            }
        
    
 constructor...
        
            function Animal() {
                this.name = "name";
                this.sound = "grrr";
                this.owner = "owner";
            }   
        
    
 like a mutator...  //can be overloaded (multple versions w/different # of args)
        
            Animal.prototype.setOwner = function(newOwner) {
                if (typeof newOwner != "undefined") {
                    this.owner = newOwner;
                } else {
                    document.write('please enter a valid owner name');
                }
            }
        
    
 like an accessor...
        
            Animal.prototype.getOwner = function() {
                return this.owner;
            }
        
    
 repeat for sound and name properties...
        
            Animal.prototype.setName = function(newName) {
                if (typeof newName != "undefined") {
                    this.name = newName;
                } else {
                    document.write('please enter a valid name');
                }
            }
       
            Animal.prototype.getName = function() {
                return this.name;
            }   
        
            Animal.prototype.setSound = function(newSound) {
                if (typeof newSound != "undefined") {
                    this.sound = newSound;
                } else {
                    document.write('please enter a valid sound');
                }
            }
        
            Animal.prototype.getSound = function() {
                return this.sound;
            }
        
    
 create a new animal...
        
            var dog = new Animal();
            dog.setName("spot");
            dog.setSound();

            document.write(dog.getName());  //"spot"
            document.write(dog.getSound());  // "grrr" because newSound was undefined
        
    
 add new method to Animal (extending object)...
        
            Animal.prototype.english = function() {
                alert('sorry, I do not speak English');
            }
        
    
 now objects defined from here on out will have inherited property of .english()...
        
            var cat = new Animal();
            cat.english   //shows method .english as an expression   
            cat.english()   // invokes function and alert box pops up 'sorry, I do not speak English'
        
    
 create new Pet class that inherits from Animal...
        
            function Pet() {
                this.home="inside house"
            }

            Pet.prototype = new Animal();

new properties/methods can be added to Pet.prototype independent of Animal
