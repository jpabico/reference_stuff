    config.use.transactional_fixtures = false
    config.infer_base_class_for_anonymous_controllers = false
    config.order = "random"

app/controllers/tasks_controller.rb

class TasksController < ApplicationController
    def index
        @tasks = Task.all
    end

    def show
        @task = Task.find(params[:id])
    end

    def new
        @task = Task.new
    end

    def create
        @task = Task.new(title: params[:task][:title])
        if @task.save
            redirect_to tasks_path
        else
            render :new
        end
    end

    def edit
        @task = Task.find(params[:id])
    end

    def update
        @task = Task.find(params[:id])
        if @task.update_atributes(title: params[:task][:title])
            redirect_to tasks_path
        else
            render :edit
        end
    end

    def destroy
        Task.delete(params[:id])
        redirect_to tasks_path
    end
end
